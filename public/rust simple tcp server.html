 1use std::{net::TcpListener, io::Read};

 3fn main() {
 4    let listener = TcpListener::bind("0.0.0.0:3333").expect("Could not bind to port");
 5
 6    for stream in listener.incoming() {
 7        match stream {
 8            Ok(mut stream) => {
 9                println!("Connection established!");
10                let mut total_buffer = String::new();
11                loop {
12                    let mut buffer = [0; 64];
13                    match stream.read(&mut buffer) {
14                        Ok(size) => {
15                            total_buffer.push_str(&String::from_utf8_lossy(&buffer[..size]));
16                            if size < 64 {
17                                break;
18                            }
19                        }
20                        Err(e) => {
21                            println!("Error: {}", e);
22                            break;
23                        }
24                    }
25                }
26                println!("{}", total_buffer);
27            }
28            Err(e) => {
29                println!("Connection failed: {}", e);
30            }
31        }
32    }
33}
