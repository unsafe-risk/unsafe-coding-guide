 1<?php
 2
 3$server = stream_socket_server("tcp://localhost:8000", $errno, $errstr);
 4
 5if ($server === false) {
 6    throw new Exception("Could not create server: $errstr ($errno)");
 7}
 8
 9for (;;) {
10    $client = stream_socket_accept($server);
11    if ($client === false) {
12        echo "Could not accept client\n";
13        continue;
14    }
15
16    $request = '';
17    while (true) {
18        $buf = fread($client, 8192);
19        $request .= $buf;
20        if (strlen($buf) < 8192) {
21            break;
22        }
23    }
24
25    $response = '';
26    $response .= $request;
27
28    $request_data = json_decode($request, true);
29    $response_data = array();
30    $response_data['name'] = $request_data['name'];
31    $response_data['greeting'] = "Hello, {$request_data['name']}!";
32
33    if (fwrite($client, json_encode($response_data)) === false) {
34        echo "Could not write response\n";
35    }
36
37    fclose($client);
