 3import (
 4	"bufio"
 5	"context"
 6	"fmt"
 7	"io"
 8	"os"
 9	"os/signal"

12	"github.com/docker/docker/api/types"
13	"github.com/docker/docker/api/types/container"
14	"github.com/docker/docker/client"
15	"github.com/docker/go-connections/nat"
16)

18func main() {
19	ctx := context.Background()
20	cli, err := client.NewClientWithOpts(client.FromEnv, client.WithAPIVersionNegotiation())
21	if err != nil {
22		panic(err)
23	}
24
25	reader, err := cli.ImagePull(ctx, "docker.io/yeasy/simple-web", types.ImagePullOptions{})
26	if err != nil {
27		panic(err)
28	}
29	defer reader.Close()
30	io.Copy(os.Stdout, reader)
31
32	hostConfig := &container.HostConfig{
33		AutoRemove: true,
34		PortBindings: nat.PortMap{
35			"80/tcp": []nat.PortBinding{
36				{
37					HostIP:   "0.0.0.0",
38					HostPort: "8080",
39				},
40			},
41		},
42	}
43
44	resp, err := cli.ContainerCreate(ctx, &container.Config{
45		Image: "yeasy/simple-web",
46		ExposedPorts: nat.PortSet{
47			"80/tcp": struct{}{},
48		},
49		Tty: false,
50	}, hostConfig, nil, nil, "")
51	if err != nil {
52		panic(err)
53	}
54
55	if err := cli.ContainerStart(ctx, resp.ID, types.ContainerStartOptions{}); err != nil {
56		panic(err)
57	}
58
59	out, err := cli.ContainerLogs(ctx, resp.ID, types.ContainerLogsOptions{ShowStdout: true})
60	if err != nil {
61		panic(err)
62	}
63
64	go func() {
65		for {
66			scanner := bufio.NewReader(out)
67			line, _, err := scanner.ReadLine()
68			if err != nil {
69				if err == io.EOF {
70					break
71				}
72				panic(err)
73			}
74			fmt.Printf("[%s-%s]: %s\n", "simple-web", resp.ID[:10], string(line[8:]))
75		}
76	}()
77
78	terminalChan := make(chan os.Signal, 1)
79	signal.Notify(terminalChan, os.Interrupt, os.Signal(syscall.SIGTERM))
80	<-terminalChan
81
82	if err := cli.ContainerStop(ctx, resp.ID, nil); err != nil {
83		panic(err)
84	}
85}
