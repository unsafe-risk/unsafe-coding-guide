 3import "sync"
 4
 5type Element[T any] struct {
 6	value T
 7	next  *Element[T]
 8}
 9
10type Queue[T any] struct {
11	head *Element[T]
12	tail *Element[T]
13	size int
14
15	pool sync.Pool
16}
17
18func New[T any]() *Queue[T] {
19	return &Queue[T]{
20		nil,
21		nil,
22		0,
23		sync.Pool{
24			New: func() interface{} {
25				return &Element[T]{}
26			},
27		},
28	}
29}
30
31func (q *Queue[T]) Push(value T) {
32	e := q.pool.Get().(*Element[T])
33	e.value = value
34	e.next = nil
35	if q.tail == nil {
36		q.head = e
37	} else {
38		q.tail.next = e
39	}
40	q.tail = e
41}
42
43func (q *Queue[T]) Pop() (value T, check bool) {
44	if q.size == 0 {
45		return value, false
46	}
47	e := q.head
48	value = e.value
49	q.pool.Put(e)
50	q.head = q.head.next
51	q.size--
52	return value, true
53}
