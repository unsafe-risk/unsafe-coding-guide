 1package main
 2
 3import (
 4	"fmt"
 5	"os"
 6	"os/exec"
 7)
 8
 9func main() {
10	if len(os.Args) < 3 {
11		fmt.Println("too few args.")
12		return
13	}
14	rootName := os.Args[1]
15	targetName := os.Args[2]
16
17	args := []string{"tar", "cvpzf", targetName}
18	excludeFlag := "--exclude="
19	excludeNames := []string{targetName, "/proc", "/lost+found", "/media", "/mnt", "/sys", "/run", "/dev"}
20	excludeNames = append(excludeNames, os.Args[3:]...)
21
22	for _, s := range excludeNames {
23		args = append(args, fmt.Sprintf("%s%s", excludeFlag, s))
24	}
25
26	cmd := exec.Command("sudo", args...)
27	cmd.Args = append(cmd.Args, rootName)
28	cmd.Stdout = os.Stdout
29	cmd.Stdin = os.Stdin
30	cmd.Stderr = os.Stderr
31	err := cmd.Run()
32	if err != nil {
33		fmt.Println(err)
34		return
35	}
36}
